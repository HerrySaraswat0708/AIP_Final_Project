%% ── SLIDE 2: Motivation ──────────────────────────────────────────────────────
\begin{frame}{Motivation \& Problem Statement}
\begin{columns}[T]
\column{0.53\textwidth}
\begin{block}{The Core Problem: Domain Shift in CLIP}
CLIP is powerful for zero-shot recognition, but suffers when the visual
distribution of a test dataset drifts from the text-embedding space:
\[
  \delta_c = 1 - \cos\!\bigl(\boldsymbol{\mu}_c^{\text{text}},\;
                              \boldsymbol{\mu}_c^{\text{img}}\bigr)
\]
We measured $\delta \approx 63$--$69\%$ across all 5 datasets.
\end{block}
\vspace{4pt}
\begin{exampleblock}{Research Questions}
\begin{enumerate}
  \item Can we adapt CLIP \emph{at test time} with no labels?
  \item \textbf{TDA} (cache-based) vs \textbf{FreeTTA} (online EM) --- which is better?
  \item \emph{Why}? Under what conditions does each method win?
\end{enumerate}
\end{exampleblock}

\small
\textbf{Our contribution:} 8 novel analyses (not in either paper) that
mathematically characterise the advantage.

%% ── SLIDE 3: Background ──────────────────────────────────────────────────────
\begin{frame}{Background: CLIP and Test-Time Adaptation}
\begin{columns}[T]
\column{0.55\textwidth}
\begin{block}{CLIP Zero-Shot Prediction}
\[
  \hat{y}_{\text{clip}} = \arg\max_c\;
    \underbrace{\bm{x} \cdot \bm{t}_c}_{\text{cosine sim}} \cdot \tfrac{1}{\tau}
  \qquad \bm{x},\bm{t}_c \in \mathbb{R}^{512}
\]
\begin{itemize}
  \item $\bm{x}$: ViT-B/16 image feature (frozen)
  \item $\bm{t}_c$: text embedding of ``a photo of a \{class\}''
  \item $\tau$: learned temperature
\end{itemize}
\end{block}
\vspace{4pt}
\begin{alertblock}{Why Zero-Shot Fails Under Domain Shift}
CLIP outputs near-\textbf{uniform} softmax on all 5 test datasets:
\[
  \bar{H}_{\text{norm}} = \frac{-\sum_c p_c\log p_c}{\log C} \approx 1.0
\]
Confidence $\approx 1/C$ --- adaptation is both necessary and hard.
\end{alertblock}

\column{0.43\textwidth}
\begin{block}{Test-Time Adaptation (TTA) Setup}
\begin{itemize}
  \item No training data or labels
  \item Backbone frozen
  \item Online streaming: one sample at a time
  \item Update auxiliary parameters only
\end{itemize}
\end{block}
\vspace{4pt}

\end{frame}

%% ── SLIDE 4: TDA ─────────────────────────────────────────────────────────────
\begin{frame}{Baseline: TDA --- Test-Time Dynamic Adapter}
\begin{columns}[T]
\column{0.54\textwidth}
\begin{block}{Core Idea}
Build an online \textbf{feature cache} from the test stream and add
cache-affinity terms to CLIP logits.
\end{block}
\vspace{4pt}
\textbf{Cache construction} (per sample $\bm{x}_i$):
\begin{enumerate}\small
  \item Compute $H_{\text{norm},i} = -\sum_c p_c\log p_c\,/\,\log C$
  \item \BL{Positive cache}: always insert $(\bm{x}_i, \hat{y}_i)$;
        evict highest-$H$ entry if at capacity (cap $= 3$/class)
  \item \HL{Negative cache}: insert \emph{only} if $0.2 < H_{\text{norm}} < 0.5$
\end{enumerate}
\vspace{4pt}
\textbf{Fused prediction:}
\[
  \bm{f} = \bm{\ell}_{\text{clip}}
         + \alpha \underbrace{\textstyle\sum_k e^{\beta\cos(\bm{x},\bm{c}^+_k)}}_{\text{pos. affinity}}
         - \alpha \underbrace{\textstyle\sum_k e^{\beta\cos(\bm{x},\bm{c}^-_k)}}_{\text{neg. affinity}}
\]

\column{0.44\textwidth}
\begin{tikzpicture}[font=\small, scale=0.85,
  box/.style={rounded corners=3pt, draw, thick, minimum width=2.1cm,
              minimum height=0.6cm, align=center, font=\footnotesize}]
  \node[box,fill=mblue!15]  (cl) at (0,3.5)  {CLIP logits $\bm{\ell}$};
  \node[box,fill=mgreen!20] (pc) at (-1.2,2) {Pos.\ cache\\ (always)};
  \node[box,fill=mred!20]   (nc) at (1.2,2)  {Neg.\ cache};
  \node[box,fill=mgold!30]  (gt) at (1.2,3)  {\footnotesize Gate\\$0.2\!<\!H\!<\!0.5$};
  \node[box,fill=mblue!25]  (ou) at (0,0.8)  {Final logits $\bm{f}$};
  \draw[->] (cl) -- (ou);
  \draw[->, mgreen!70!black, thick] (pc) -- node[left,font=\tiny]{$+\alpha$} (ou);
  \draw[->, mred!70!black,   thick] (nc) -- node[right,font=\tiny]{$-\alpha$} (ou);
  \draw[->] (gt) -- (nc);
\end{tikzpicture}
\vspace{4pt}
\begin{alertblock}{Critical Weakness Found}
\small When $H_{\text{norm}} \approx 1.0$ (all 5 datasets), the negative-cache
gate \textbf{never fires} $\Rightarrow$ negative cache stays \textbf{empty}.
Positive cache saturates at $C\!\times\!3$ slots, then \textbf{frozen}.
\end{alertblock}
\end{columns}
\end{frame}

%% ── SLIDE 5: FreeTTA ─────────────────────────────────────────────────────────
\begin{frame}{Extension: FreeTTA --- Free Online EM Adaptation}
\begin{columns}[T]
\column{0.53\textwidth}
\begin{block}{Core Idea}
Model features as class-conditional Gaussians
$p(\bm{x}|y{=}c) = \mathcal{N}(\bm{\mu}_c, \sigma^2 I)$
and estimate $\bm{\mu}_c$ via \emph{soft online EM}.
\end{block}
\vspace{4pt}
\textbf{E-step (predict):}
\[
  \ell^{\text{gen}}_c = -\tfrac{\|\bm{x}-\bm{\mu}_c\|^2}{2\sigma^2},\quad
  \hat{\bm{p}} = \text{softmax}\!\bigl(\bm{\ell}_{\text{clip}} + \alpha\bm{\ell}^{\text{gen}}\bigr)
\]
\textbf{M-step (update) with soft gate:}
\[
  w_i = e^{-\beta H_{\text{norm},i}} \in (0,1]
  \quad\text{(\emph{always} positive!)}
\]
\[
  N_{y,c} \mathrel{+}= w_i\hat{p}_{ic},\qquad
  \bm{\mu}_c \mathrel{+}= \frac{w_i\hat{p}_{ic}}{N_{y,c}}(\bm{x}_i - \bm{\mu}_c)
\]
Initialise: $\bm{\mu}_c(0) = \bm{t}_c$ (CLIP text prototype)

\column{0.45\textwidth}
\begin{tikzpicture}[font=\small, scale=0.85,
  box/.style={rounded corners=3pt, draw, thick, minimum width=2.2cm,
              minimum height=0.6cm, align=center, font=\footnotesize}]
  \node[box,fill=mblue!15]  (cl) at (0,3.8) {CLIP logits $\bm{\ell}$};
  \node[box,fill=mred!15]   (ge) at (0,2.8) {Gen.\ logits $\bm{\ell}^{\text{gen}}$};
  \node[box,fill=mgreen!20] (mu) at (-1.3,1.7){Class means $\bm{\mu}_c$};
  \node[box,fill=mgold!30]  (wt) at (1.3,1.7){Soft gate $w\!=\!e^{-\beta H}$};
  \node[box,fill=mblue!25]  (pr) at (0,0.6) {Fused predict $\hat{\bm{p}}$};
  \node[box,fill=mred!25]   (up) at (0,-0.5){M-step: update $\bm{\mu}_c$};
  \draw[->] (cl) -- (ge);
  \draw[->] (ge) -- (pr);
  \draw[->] (mu) -- (ge);
  \draw[->] (pr) -- (up);
  \draw[->] (wt) -- (up);
  \draw[->, dashed, mgreen!70!black, thick]
    (up.west) to[bend right=50] (mu.south);
\end{tikzpicture}
\vspace{4pt}
\begin{exampleblock}{Key Advantage}
\small $w_i > 0$ \textbf{always}. Even maximally uncertain samples
contribute $e^{-3} \approx 5\%$ signal. No saturation. No empty cache.
\end{exampleblock}
\end{columns}
\end{frame}

%% ── SLIDE 9: Prediction Change ───────────────────────────────────────────────
\begin{frame}{Section 1 --- Prediction Change Analysis}
\begin{columns}[T]
\column{0.5\textwidth}
\begin{block}{Metrics Defined}
For each sample $i$ and method $m$:
\begin{align*}
  \text{BF}_i &= \mathbf{1}[\hat{y}_{\text{clip}}\!\ne\! y_i,\;\hat{y}_m\!=\! y_i]\\
  \text{HF}_i &= \mathbf{1}[\hat{y}_{\text{clip}}\!=\! y_i,\;\hat{y}_m\!\ne\! y_i]\\
  \text{Flip Precision} &= \frac{\text{BF}}{\text{BF}+\text{HF}}
\end{align*}
Precision $> 0.5$ means method \emph{improves} more than it harms.
\end{block}

%% ── SLIDE 10: Entropy Analysis ───────────────────────────────────────────────
\begin{frame}{Sections 2 \& 5 --- Entropy Analysis \& Gate Efficiency}
\begin{columns}[T]
\column{0.5\textwidth}
\begin{alertblock}{Critical Discovery}
CLIP assigns near-\textbf{uniform} probabilities across all 5 datasets:
\[
  \bar{H}_{\text{norm}} \approx 1.0 \quad\forall\text{ dataset}
\]
\textbf{Consequence for TDA:}
\begin{itemize}
  \item Negative-cache gate: $P(0.2 < H < 0.5) = \mathbf{0\%}$
  \item Negative cache stays \textbf{completely empty}
\end{itemize}
\textbf{Consequence for FreeTTA:}
\begin{itemize}
  \item Soft gate: $w = e^{-\beta\cdot 1} = e^{-3} \approx \mathbf{5\%}$
  \item Always positive --- accumulates over all $N$ samples
\end{itemize}
\end{alertblock}
\vspace{2pt}
{\small Total adaptation signal: TDA neg-cache $= 0$ events;\;
FreeTTA $\approx 0.05N$ weighted events.}

\column{0.48\textwidth}
\begin{block}{Post-Adaptation Entropy (Caltech-101)}

%% ── SLIDE 14: When Each Method Wins ──────────────────────────────────────────
\begin{frame}{Mathematical Conditions: When Does Each Method Win?}
\begin{columns}[T]
\column{0.5\textwidth}
\begin{exampleblock}{FreeTTA Always Wins When:}
\begin{enumerate}\small
  \item \textbf{High domain shift} ($\delta_c > 0.15$)\\
        $\bm{\mu}_c(t)$ adapts toward image centroids;
        TDA cache exemplars cannot correct the text-image gap.
  \item \textbf{Few classes} ($C < 50$)\\
        TDA saturates at $C\!\times\!3$ slots in $<\!2\%$ of stream.
        FreeTTA: no saturation.
  \item \textbf{High CLIP entropy} ($\bar{H}_\text{norm} > 0.8$)\\
        TDA neg-gate fires 0\%; FreeTTA soft gate always $> 0$.
  \item \textbf{Long streams / late phase} (Q4)\\
        Accumulated FreeTTA adaptation $\gg$ static TDA cache ($+$3.5\%).
  \item \textbf{Rare/imbalanced classes}\\
        TDA cache empty for classes with $<3$ test samples.
\end{enumerate}
\end{exampleblock}

\column{0.48\textwidth}
\begin{alertblock}{TDA Can Win When:}
\begin{enumerate}\small
  \item \textbf{Low shift + high CLIP confidence}\\
        Cache exemplars closely match query (Pets/ImageNet).
  \item \textbf{Large $C$ + abundant samples/class}\\
        Cache fills richly; affinity retrieval precise.
  \item \textbf{Short streams / early phase} (Q1--Q2)\\
        TDA provides immediate boost before FreeTTA converges.
  \item \textbf{Medium-entropy stream} ($H\in(0.2, 0.5)$)\\
        Neg-cache gate fires; correction signal available.
\end{enumerate}
\end{alertblock}
\vspace{6pt}
\begin{block}{Structural Insight}
\small \BL{TDA}: memory-bounded retriever.
$\;$ \HL{FreeTTA}: statistically consistent online learner.
As $N\!\to\!\infty$: $\bm{\mu}_c(t) \to \bm{\mu}_c^{\text{img}}$.
TDA cannot improve beyond cache capacity.
\end{block}
\end{columns}
\end{frame}

%% ── SLIDE 15: Conclusion ─────────────────────────────────────────────────────
\begin{frame}{Conclusion \& Future Work}
\begin{columns}[T]
\column{0.53\textwidth}
\begin{block}{Summary of Findings}
\begin{itemize}
  \item FreeTTA outperforms TDA on \textbf{3/5} datasets; avg $\Delta = +$1.48\%
  \item \textbf{EuroSAT} decisive: $+$6.01\% from high shift ($C=10$) and early saturation
  \item CLIP is \textbf{maximally uncertain} ($H_\text{norm}\approx 1$) on all datasets,
        structurally starving TDA's negative cache
  \item FreeTTA soft gate \textbf{always adapts} ($w\!>\!0$); TDA gate fires \textbf{0 times}
  \item FreeTTA uniquely rescues 771 samples;
        unique harm on only 178 (vs.\ TDA's 523)
  \item FreeTTA advantage \textbf{grows} with stream: $+$2.1\% (Q1) $\to$ $+$3.5\% (Q4)
  \item Domain shift correlation with accuracy gap: Pearson $r = +0.46$
\end{itemize}
\end{block}

\column{0.45\textwidth}
\begin{exampleblock}{Future Directions}
\begin{itemize}\small
  \item \textbf{Hybrid:} warm-start FreeTTA $\bm{\mu}_c(0)$ from TDA cache exemplars
  \item \textbf{Adaptive $\beta$:} schedule entropy gate threshold as stream confidence rises
  \item \textbf{Per-class saturation detection:} freeze TDA cache slots on convergence
  \item \textbf{Harder benchmarks:} WILDS, medical imaging, OOD detection
  \item \textbf{Multi-modal adaptation:} update text embeddings jointly
\end{itemize}
\end{exampleblock}
