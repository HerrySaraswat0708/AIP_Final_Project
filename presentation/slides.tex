\begin{frame}{Motivation \& Problem Statement}
\begin{columns}[T]
\column{0.52\textwidth}
  \begin{block}{The Core Problem}
    CLIP is powerful at zero-shot recognition but suffers from \textbf{domain shift}:
    text embeddings $\bm{t}_c$ do not align perfectly with the visual distribution
    of a \emph{specific} test domain.
    \[
      \delta_c = 1 - \cos(\bm{\mu}^{\text{text}}_c,\; \bm{\mu}^{\text{img}}_c)
    \]
    We measured $\delta \approx 63$--$69\%$ across all 5 datasets.
  \end{block}
  \vspace{4pt}
  \begin{exampleblock}{Research Questions}
    \begin{enumerate}
      \item Can we adapt CLIP \emph{at test time} without labels?
      \item Which adaptation strategy is more effective?
      \item \textbf{Why} does one outperform the other?
    \end{enumerate}
  \end{exampleblock}

% ── Slide 3: Background ───────────────────────────────────────────────────────
\begin{frame}{Background: CLIP and Test-Time Adaptation}
\begin{columns}[T]
\column{0.56\textwidth}
  \begin{block}{CLIP Zero-Shot Classification}
    \[
      \hat{y}_{\text{clip}} = \arg\max_c\;
        \frac{\bm{x} \cdot \bm{t}_c}{\|\bm{x}\|\|\bm{t}_c\|} \cdot \frac{1}{\tau}
    \]
    \begin{itemize}
      \item $\bm{x} \in \mathbb{R}^{512}$: ViT-B/16 image feature
      \item $\bm{t}_c$: text embedding of \texttt{"a photo of a \{class\}"}
      \item $\tau$: learned temperature
    \end{itemize}
  \end{block}
  \vspace{4pt}
  \begin{alertblock}{Why Zero-Shot Fails Under Distribution Shift}
    CLIP assigns \textbf{near-uniform} probabilities on uncertain domains:
    \[
      \bar{H}_{\text{norm}} = \frac{H}{\log C} \approx 1.0 \quad
      \text{(measured on all 5 datasets)}
    \]
    Confidence $\approx 1/C$, so adaptation is hard but necessary.
  \end{alertblock}

\column{0.42\textwidth}
  \begin{block}{Test-Time Adaptation (TTA)}
    \textbf{Goal:} Improve predictions \emph{on-the-fly}, one test sample at a time,
    with no access to labels.
    \vspace{4pt}
    \begin{itemize}
      \item No training data reused
      \item No label supervision
      \item Model backbone frozen
      \item Must be online (streaming)
    \end{itemize}
  \end{block}
  \vspace{6pt}
  \begin{table}
    \small
    \begin{tabular}{lcc}
    \toprule
    Dataset & $C$ & $N_{\text{test}}$\\
    \midrule
    Caltech-101   & 101 & 2{,}465\\
    DTD           & 47  & 1{,}880\\
    EuroSAT       & 10  & 8{,}100\\
    Oxford Pets   & 37  & 3{,}669\\
    ImageNetV2    & 1000 & 10{,}000\\
    \bottomrule
    \end{tabular}
  \end{table}
\end{columns}
\end{frame}

% ── Slide 4: TDA Algorithm ────────────────────────────────────────────────────
\begin{frame}{Baseline Method: TDA (Test-Time Dynamic Adapter)}
\begin{columns}[T]
\column{0.52\textwidth}
  \begin{block}{Core Idea}
    Build an online \textbf{feature cache} from the test stream and add
    cache-affinity terms to CLIP logits.
  \end{block}
  \vspace{4pt}
  \textbf{Cache construction} (per test sample $\bm{x}_i$):
  \begin{enumerate}\small
    \item Compute $H_{\text{norm},i} = -\sum_c p_c \log p_c \,/\, \log C$
    \item \blue{Positive cache}: always insert $(\bm{x}_i, \hat{y}_i)$;
          evict highest-$H$ entry if full (cap $= 3$/class)
    \item \highlight{Negative cache}: insert only if $0.2 < H_{\text{norm}} < 0.5$
  \end{enumerate}
  \vspace{4pt}
  \textbf{Prediction:}
  \[
    \bm{f} = \underbrace{\bm{\ell}_{\text{clip}}}_{\text{base}}
           + \alpha \underbrace{\sum_k e^{\beta \cos(\bm{x},\bm{c}^+_k)}}_{\text{pos. affinity}}
           - \alpha \underbrace{\sum_k e^{\beta \cos(\bm{x},\bm{c}^-_k)}}_{\text{neg. affinity}}
  \]

\column{0.46\textwidth}
  \begin{tikzpicture}[scale=0.82,
    box/.style={rounded corners=3pt, draw, thick, minimum width=2.2cm,
                minimum height=0.55cm, align=center, font=\footnotesize}]
    \node[box, fill=primary!15] (clip) at (0,3) {CLIP logits $\bm{\ell}$};
    \node[box, fill=green2!20]  (pos)  at (-1.3,1.6) {Pos cache};
    \node[box, fill=accent!20]  (neg)  at (1.3,1.6)  {Neg cache};
    \node[box, fill=gold!30]    (gate) at (1.3,2.5)  {Gate: $0.2\!<\!H\!<\!0.5$};
    \node[box, fill=primary!25] (out)  at (0,0.3) {Final logits $\bm{f}$};
    \draw[->] (clip) -- (out);
    \draw[->] (pos)  -- node[left,  font=\tiny]{$+\alpha$} (out);
    \draw[->] (neg)  -- node[right, font=\tiny]{$-\alpha$} (out);
    \draw[->] (gate) -- (neg);
  \end{tikzpicture}
  \vspace{6pt}
  \begin{alertblock}{Key Weakness Discovered}
    \small When $H_{\text{norm}} \approx 1.0$ (high CLIP uncertainty),
    the negative-cache gate admits \textbf{0\% of samples} $\Rightarrow$
    negative cache empty throughout test.
  \end{alertblock}
\end{columns}
\end{frame}

% ── Slide 5: FreeTTA Algorithm ────────────────────────────────────────────────
\begin{frame}{Extension Method: FreeTTA (Online EM Adaptation)}
\begin{columns}[T]
\column{0.52\textwidth}
  \begin{block}{Core Idea}
    Model image features as class-conditional Gaussians
    $p(\bm{x}|y{=}c) = \mathcal{N}(\bm{x};\,\bm{\mu}_c,\sigma^2\mathbf{I})$
    and estimate $\bm{\mu}_c$ via soft online EM.
  \end{block}
  \vspace{4pt}
  \textbf{E-step (predict):}
  \[
    \bm{\ell}^{\text{gen}}_c = -\frac{\|\bm{x}-\bm{\mu}_c\|^2}{2\sigma^2},\quad
    \bm{f} = \bm{\ell}_{\text{clip}} + \alpha\,\bm{\ell}^{\text{gen}}
  \]
  \textbf{M-step (update) — soft gate:}
  \[
    w_i = e^{-\beta H_{\text{norm},i}} \in (0,1]
  \]
  \[
    N_{y,c} \mathrel{+}= w_i\,\hat{p}_{ic},\quad
    \bm{\mu}_c \mathrel{+}= \frac{w_i\,\hat{p}_{ic}}{N_{y,c}}(\bm{x}_i - \bm{\mu}_c)
  \]
  \vspace{2pt}
  \small Init: $\bm{\mu}_c(0) = \bm{t}_c$ (CLIP text prototype)

\column{0.46\textwidth}
  \begin{tikzpicture}[scale=0.82,
    box/.style={rounded corners=3pt, draw, thick, minimum width=2.2cm,
                minimum height=0.55cm, align=center, font=\footnotesize}]
    \node[box, fill=primary!15]  (clip) at (0,3.5)  {CLIP logits $\bm{\ell}$};
    \node[box, fill=accent!15]   (gen)  at (0,2.5)  {Gen logits $\bm{\ell}^{\text{gen}}$};
    \node[box, fill=green2!20]   (mu)   at (-1.4,1.5){Class means $\bm{\mu}_c$};
    \node[box, fill=gold!30]     (gate) at (1.4,1.5) {Soft gate $w = e^{-\beta H}$};
    \node[box, fill=primary!25]  (out)  at (0,0.3)  {Fused predict};
    \node[box, fill=accent!25]   (upd)  at (0,-0.7) {M-step: update $\bm{\mu}_c$};
    \draw[->] (clip) -- (gen);
    \draw[->] (gen)  -- (out);
    \draw[->] (mu)   -- (gen);
    \draw[->] (out)  -- (upd);
    \draw[->] (gate) -- (upd);
    \draw[->, dashed, green2] (upd.west) to[bend right=50] (mu.south);
  \end{tikzpicture}
  \vspace{4pt}
  \begin{exampleblock}{Key Advantage}
    \small Soft gate $w > 0$ \textbf{always} — even maximally uncertain
    samples contribute $\approx e^{-3} \approx 5\%$ signal.
    No saturation. No empty cache.
  \end{exampleblock}
\end{columns}
\end{frame}

% ── Slide 9: Prediction Change Analysis ──────────────────────────────────────
\begin{frame}{Section 1 — Prediction Change Analysis}
\begin{columns}[T]
\column{0.48\textwidth}
  \begin{block}{Methodology}
    For each method $m \in \{\text{TDA, FreeTTA}\}$ and test sample $i$:
    \[
      \text{BF}_i = \mathbb{1}[\hat{y}_{\text{clip}}\!\ne\! y,\; \hat{y}_m\!=\! y]
    \]
    \[
      \text{HF}_i = \mathbb{1}[\hat{y}_{\text{clip}}\!= y,\; \hat{y}_m\!\ne\! y]
    \]
    \[
      \text{Flip Precision} = \frac{\text{BF}}{\text{BF}+\text{HF}}
    \]
  \end{block}
  \vspace{4pt}

% ── Slide 14: When Each Method Wins ───────────────────────────────────────────
\begin{frame}{Mathematical Conditions: When Each Method Wins}
\begin{columns}[T]
\column{0.5\textwidth}
  \begin{exampleblock}{\faArrowUp\ FreeTTA Always Wins When:}
    \begin{enumerate}\small
      \item \textbf{High domain shift}: $\delta_c > 0.15$\\
            $\bm{\mu}_c(t)$ adapts; TDA cache stagnates
      \item \textbf{Few classes} ($C < 50$):\\
            TDA saturates in $< 2\%$ of stream at $C \times 3$ slots
      \item \textbf{High CLIP entropy} ($\bar{H}_{\text{norm}} > 0.8$):\\
            TDA neg-cache gate fires 0\%; FreeTTA soft gate $> 0$ always
      \item \textbf{Long test streams / late phase}:\\
            FreeTTA accumulated adaptation $\gg$ TDA static cache (Q4: +3.5\%)
      \item \textbf{Rare / imbalanced classes}:\\
            TDA cache empty for classes with $<$ 3 test samples
    \end{enumerate}
  \end{exampleblock}

\column{0.48\textwidth}
  \begin{alertblock}{\faArrowDown\ TDA Can Win When:}
    \begin{enumerate}\small
      \item \textbf{Low domain shift + high CLIP confidence}:\\
            Cache exemplars closely match query; Pets/ImageNet regime
      \item \textbf{Large $C$ + abundant samples/class}:\\
            Cache fills richly; affinity retrieval precise (ImageNet: $C=1000$)
      \item \textbf{Short streams (early phase Q1--Q2)}:\\
            TDA provides immediate boost before FreeTTA converges
      \item \textbf{Medium-entropy streams} ($H_{\text{norm}} \in (0.2, 0.5)$):\\
            Neg-cache gate fires; correction signal available for TDA
    \end{enumerate}
  \end{alertblock}
  \vspace{6pt}
  \begin{block}{\small Structural Insight}
    \small TDA is a \textbf{memory-bounded} retriever.
    FreeTTA is a \textbf{statistically-consistent} online learner.
    As $N \to \infty$: $\bm{\mu}_c(t) \to \bm{\mu}_c^{\text{img}}$ (FreeTTA converges).
    TDA cannot improve beyond its cache capacity.
  \end{block}
\end{columns}
\end{frame}

% ── Slide 15: Conclusion ──────────────────────────────────────────────────────
\begin{frame}{Conclusion \& Future Work}
\begin{columns}[T]
\column{0.52\textwidth}
  \begin{block}{Summary of Findings}
    \begin{itemize}
      \item FreeTTA outperforms TDA on \textbf{3/5} datasets; average $\Delta = +1.48\%$
      \item \textbf{EuroSAT} is the decisive dataset: +6.01\% due to high domain shift
            and early cache saturation ($C=10$)
      \item CLIP is \textbf{maximally uncertain} on all test domains: $H_{\text{norm}} \approx 1.0$,
            which structurally starves TDA's negative cache
      \item FreeTTA's soft gate \textbf{always adapts}; TDA's hard gate \textbf{never fires}
            in the high-entropy regime
      \item FreeTTA uniquely rescues 771 samples; causes unique harm on only 178
      \item FreeTTA advantage \textbf{grows with stream length}: strongest in Q4 (+3.5\%)
    \end{itemize}
  \end{block}

\column{0.46\textwidth}
  \begin{exampleblock}{Future Work}
    \begin{itemize}\small
      \item Combine TDA cache memory with FreeTTA's generative model
            (warm-start $\bm{\mu}_c$ from cache exemplars)
      \item Adaptive $\beta$ scheduling: reduce entropy gate threshold as
            stream progresses and model becomes more confident
      \item Per-class saturation detection: dynamically freeze TDA cache
            slots when convergence is detected
      \item Test on longer/harder streams: WILDS benchmarks,
            medical imaging domains
    \end{itemize}
  \end{exampleblock}
  \vspace{6pt}
  \begin{alertblock}{\centering\large Final Verdict}
    \centering
    \textbf{FreeTTA $>$ TDA} in high-uncertainty, high-shift domains.\\
    TDA $\approx$ FreeTTA in low-shift, large-class settings.
  \end{alertblock}
