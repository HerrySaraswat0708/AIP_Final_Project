\begin{frame}{Motivation: Why Test-Time Adaptation?}
\begin{columns}[T]
\column{0.50\textwidth}
  \begin{block}{The Core Problem}
    \begin{itemize}\small
      \item CLIP is trained on web-scale data but \textbf{deployed on specialist domains} (satellite images, medical textures, pet breeds)
      \item At test time: no labels, no gradient, frozen backbone
      \item CLIP logits are nearly uniform ($H_{\rm norm} \approx 1.0$) on all 5 benchmarks — the model is maximally uncertain
      \item \textbf{Goal}: adapt predictions online using only the test stream itself
    \end{itemize}
  \end{block}
  \begin{exampleblock}{\small What Both Papers Do}
    \small Operate on \textbf{frozen CLIP features} only. No retraining, no labels, no augmentation. Pure inference-time adaptation.
  \end{exampleblock}
\column{0.48\textwidth}
  \begin{alertblock}{Why Compare TDA vs FreeTTA?}
    \begin{itemize}\small
      \item \textbf{TDA} (baseline): memory-based — stores confident exemplars in a cache and retrieves them for each query
      \item \textbf{FreeTTA} (extension): statistics-based — maintains running class-mean estimates updated via EM
      \item Both improve on CLIP, but via \textbf{fundamentally different mechanisms}
      \item We design 10 metrics and 6 new experiments to understand \emph{when and why} each wins
    \end{itemize}
  \end{alertblock}
\end{columns}
\end{frame}

%% ─────────────────────────────────────────────────────────────────────────────
%% SLIDE 3 – TDA ARCHITECTURE
%% ─────────────────────────────────────────────────────────────────────────────
\begin{frame}{Baseline Method: TDA (Test-Time Dynamic Adapter)}
\begin{columns}[T]
\column{0.52\textwidth}
  \begin{block}{Algorithm (per test sample)}
    \small
    \textbf{1.\ Compute} CLIP logits $\mathbf{l}_{\rm clip} = \tau\cdot\hat{\mathbf{x}}^\top\mathbf{W}$
    \\[4pt]
    \textbf{2.\ Route by entropy} $H_{\rm norm} = H / \log C$:\\
    \quad $H_{\rm norm} < \theta_{\rm low}$: add to \textcolor{primary}{\textbf{positive cache}}\\
    \quad $\theta_{\rm low} < H_{\rm norm} < \theta_{\rm high}$: also add to \textcolor{accent}{\textbf{negative cache}}
    \\[4pt]
    \textbf{3.\ Cache affinity} (vectorised):
    \[
      \text{aff}_{c,s} = \hat{\mathbf{x}}^\top \hat{\mathbf{f}}_{c,s}
    \]
    \textbf{4.\ Fused logits}:
    \[
      \mathbf{l}_{\rm TDA} = \mathbf{l}_{\rm clip} + \alpha\cdot\mathbf{l}^+ - \alpha'\cdot\mathbf{l}^-
    \]
  \end{block}
\column{0.46\textwidth}
  \begin{exampleblock}{Key Properties}
    \begin{itemize}\small
      \item Positive cache: $C \times K$ slots (e.g.\ $100 \times 3 = 300$ for Caltech)
      \item Negative cache gate: only fires when $H \in (0.2, 0.5)$ — \textbf{never fires} when CLIP is maximally uncertain
      \item Cache fills within first \textbf{20\% of stream} — then only eviction-replacement
      \item Memory: $O(C \cdot K \cdot D)$; Time: $O(C \cdot K \cdot D)$ per sample
    \end{itemize}
  \end{exampleblock}
  \begin{alertblock}{\small Structural Weakness}
    \small Fixed capacity $C \times K$. After saturation, TDA cannot improve further without displacing existing entries. Learning rate $\to 0$ as cache fills.
  \end{alertblock}
\end{columns}
\end{frame}

%% ─────────────────────────────────────────────────────────────────────────────
%% SLIDE 4 – FREETTA ARCHITECTURE
%% ─────────────────────────────────────────────────────────────────────────────
\begin{frame}{Extension Method: FreeTTA (Online EM Adaptation)}
\begin{columns}[T]
\column{0.52\textwidth}
  \begin{block}{EM Algorithm (per test sample)}
    \small
    \textbf{E-step} (predict with adapted means):
    \[
      \hat{y} = \arg\max_c \;\bigl[\mathbf{l}_{\rm clip} + \alpha\,(\hat{\mathbf{x}}^\top \mub_c)\bigr] \cdot \tau
    \]
    \textbf{Entropy gate}:
    \[
      \alpha_t = \exp(-\beta\cdot H_{\rm norm}) \in (0, 1]
    \]
    \textbf{M-step} (update class means):
    \[
      \mub_c \leftarrow \frac{N_c\,\mub_c + \alpha_t\, p_c\, \hat{\mathbf{x}}}{N_c + \alpha_t\, p_c}
    \]
    \[
      N_c \leftarrow N_c + \alpha_t\, p_c
    \]
  \end{block}
\column{0.46\textwidth}
  \begin{exampleblock}{Key Properties}
    \begin{itemize}\small
      \item Initialized at CLIP text embeddings $\mub_c^{(0)} = \hat{\mathbf{t}}_c$
      \item Soft gate: $\alpha_t > 0$ always — \textbf{never structurally disabled}
      \item Memory: $O(C \cdot D)$ — 15$\times$ less than TDA on ImageNet
      \item Convergence: $\mub_c(t) \to \bar{\mathbf{x}}_c$ as $N \to \infty$
      \item Time: $O(C \cdot D)$ per sample — same as CLIP inference
    \end{itemize}
  \end{exampleblock}
  \begin{alertblock}{\small Structural Advantage}
    \small No capacity limit. Gain grows monotonically with stream length. Mean EM weight decays gracefully — even $\alpha_t = 0.2$ on EuroSAT accumulates over 8,100 samples.
  \end{alertblock}
\end{columns}
\end{frame}

\begin{frame}{Entropy \& Uncertainty Analysis (Section 4)}
\begin{columns}[T]
\column{0.50\textwidth}
  \begin{block}{CLIP Is Maximally Uncertain Everywhere}
    \small Normalised CLIP entropy $H/\log C$ on all datasets:
    \begin{itemize}

    Soft gate $\alpha_t = e^{-\beta H_{\rm norm}}$ is always $> 0$ — never disabled.
  \end{block}

    \end{tabular}

    \begin{itemize}
      \item Centroid drift in early stream before means stabilize
      \item Self-corrects after $\sim$100 samples
      \item Most common on EuroSAT (0.36\%)
    \end{itemize}
  \end{exampleblock}
  \begin{alertblock}{\small Key Finding}
    \small \textbf{13.3\% of EuroSAT} is all-wrong — genuinely ambiguous terrain types (crop vs crop). Neither method can rescue these. They reflect CLIP's fundamental limitation.
  \end{alertblock}
\end{columns}

\begin{frame}{When Each Method Wins: A Decision Framework}
\begin{columns}[T]
\column{0.50\textwidth}
  \begin{exampleblock}{\faArrowUp\ FreeTTA Always Wins When:}
    \begin{enumerate}\small
      \item \textbf{High domain shift} ($\delta_c > 0.15$): $\mub_c(t)$ adapts; TDA cache stagnates
      \item \textbf{Few classes} ($C < 50$): TDA saturates in $< 2\%$ of stream at $C \times 3$ slots
      \item \textbf{High CLIP entropy} ($\bar{H}_{\rm norm} > 0.8$): TDA neg-cache gate fires 0\%; FreeTTA soft gate always active
      \item \textbf{Long streams}: FreeTTA accumulated adaptation $\gg$ TDA static cache; +31\% late-phase lift on EuroSAT
      \item \textbf{Memory-constrained deployment}: FreeTTA uses 15$\times$ less memory on ImageNet
    \end{enumerate}
  \end{exampleblock}
\column{0.48\textwidth}
  \begin{alertblock}{\faArrowDown\ TDA Can Win When:}
    \begin{enumerate}\small
      \item \textbf{Low domain shift + fine-grained} (Pets, ImageNet): cache exemplars match query geometry precisely
      \item \textbf{Cache not saturated} (early stream): retrieval is still precise
      \item \textbf{Medium entropy} ($H_{\rm norm} \in [0.2, 0.5]$): neg-cache gate fires; correction signal available
      \item \textbf{Very short streams}: TDA boosts immediately; FreeTTA needs time to converge
    \end{enumerate}
  \end{alertblock}
  \vspace{6pt}
  \begin{block}{\small Structural Insight}
    \small TDA is a \textbf{bounded memory retriever} — capacity $= C \times K$, fixed.\\
    FreeTTA is a \textbf{convergent statistical learner} — improves as $N \to \infty$.\\
    As stream grows: FreeTTA dominates.
  \end{block}
\end{columns}
\end{frame}

    \end{itemize}
